1   package mtd.model.score;
2   
3   /**
4    * Provides functionality to keep track of the player's score.
5    *
6    * @author Jack Sheridan
7    */
8   public class ScoreTracker {
9   
10      /**
11       * Player's current score.
12       */
13      private int score;
14      /**
15       * Highest score this session.
16       */
17      private int highestScore;
18  
19      /**
20       * Creates a new ScoreTracker. Sets score values to zero.
21       */
22      public ScoreTracker() {
23          this.score = 0;
24          this.highestScore = 0;
25      }
26  
27      /**
28       * Add points to player's score.
29       *
30       * @param points the amount of points to add.
31       */
32      public void addPointsToScore(int points) {
33          this.score += points;
34          evaluateHighScore();
35      }
36  
37      public int getScore() {
38          return this.score;
39      }
40  
41      /**
42       * Checks if the current score is the highest score this session.
43       *
44       * @return True if this score is also the highest score.
45       */
46      public boolean currentScoreIsHighestScore() {
47          evaluateHighScore();
48          return (this.score == this.highestScore);
49      }
50  
51      private void evaluateHighScore() {
52          if (this.score > this.highestScore) {
53              this.highestScore = this.score;
54          }
55      }
56  
57      /**
58       * Reset the player's score to 0.
59       */
60      public void reset() {
61          this.score = 0;
62      }
63  
64  }
