6   package mtd.model.models;
7   
8   import java.util.ArrayList;
9   import java.util.Collections;
10  import java.util.List;
11  import mtd.model.models.answer.Answer;
12  import mtd.model.models.answer.CorrectAnswer;
13  import mtd.model.models.answer.RandomAnswer;
14  
15  /**
16   *
17   * @author Jack Sheridan
18   */
19  /**
20   * Representation of a game question, containing the Event, one CorrectAnswer,
21   * List<RandomAnswer> of length 3, and a List<Answer> which contains
22   * CorrectAnswer and all RandomAnswers.
23   */
24  public class Question {
25  
26      private final Event event;
27      private final CorrectAnswer correctAnswer;
28      private final List<RandomAnswer> randomAnswers;
29      private List<Answer> shuffled;
30  
31      public Question(Event event, CorrectAnswer correctAnswer, List<RandomAnswer> randomAnswers) {
32          this.event = event;
33          this.correctAnswer = correctAnswer;
34          this.randomAnswers = randomAnswers;
35          shuffled = createShuffledAnswers();
36  
37      }
38  
39      public final Event getEvent() {
40          return event;
41      }
42  
43      public final CorrectAnswer getCorrectAnswer() {
44          return correctAnswer;
45      }
46  
47      public final List<RandomAnswer> getRandomAnswers() {
48          return randomAnswers;
49      }
50  
51      private List<Answer> createShuffledAnswers() {
52          List<Answer> answers = new ArrayList<Answer>();
53          answers.addAll(this.randomAnswers);
54          answers.add(correctAnswer);
55          Collections.shuffle(answers);
56          return answers;
57      }
58  
59      public List<Answer> getShuffled() {
60          return shuffled;
61      }
62  
63  }
