1   package mtd.model.create;

3   import java.util.ArrayList;
4   import java.util.List;
5   import mtd.model.load.EventLoader;
6   import mtd.model.load.SettingsLoader;
7   import mtd.model.models.Event;
8   import org.json.JSONException;
9   import org.json.JSONObject;
10  
11  /**
12   *
13   * @author Jack Sheridan
14   */
15  public class EventCreator {
16  
17      private final int quantity;
18      private final JSONObject eventsJSON;
19      private final EventPicker eventPicker;
20      private final List<Event> processedEvents;
21  
22      public EventCreator() {
23          this.quantity = new SettingsLoader().getSetting("number_of_events");
24  
25          eventsJSON = new EventLoader().getJSONRoot().getJSONObject("events");
26          processedEvents = new ArrayList<>();
27          eventPicker = new EventPicker(eventsJSON);
28      }
29  
30      /**
31       * Creates a list of events.
32       *
33       * @return List of Event objects.
34       */
35      public List<Event> createEvents() {
36          return handleEventCreationLoop();
37      }
38  
39      private List<Event> handleEventCreationLoop() {
40          ArrayList<Event> listOfEvents = new ArrayList<>();
41          for (int i = 0; i < quantity; i++) {
42              String key = eventPicker.selectEventNotYetSelected();
43  
44              Event event = createIndividualEventObject(key);
45              listOfEvents.add(event);
46          }
47          return listOfEvents;
48      }
49  
50      private Event createIndividualEventObject(String key) {
51          try {
52              JSONObject event = this.eventsJSON.getJSONObject(key);
53              Integer year = event.getInt("year");
54              String desc = event.getString("desc");
55              return new Event(year, desc);
56          } catch (JSONException e) {
57              System.out.println(e.getMessage());
58  
59          }
60          throw new IllegalStateException("event object creation failed");
61  
62      }
63  }
