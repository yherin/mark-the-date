6   package mtd.model.load;
7   
8   import java.util.HashMap;
9   import mtd.view.error.ErrorInformer;
10  import org.json.JSONException;
11  import org.json.JSONObject;
12  
13  /**
14   * Reads JSON data from events.json.
15   *
16   * @author Jack Sheridan
17   */
18  public final class SettingsLoader extends Loader {
19  
20      private static final HashMap<String, Integer> SETTINGS = new HashMap<>();
21  
22      /**
23       * Creates a new instance of SettingsLoader.
24       *
25       * @see Loader
26       */
27      public SettingsLoader() {
28          super();
29      }
30  
31      /**
32       * Reads the setting stored at the specified key in the settings.json file.
33       *
34       * @param key of the setting to be read.
35       * @return value corresponding to key.
36       */
37      public final int getSetting(String key) {
38          try {
39              if (!SETTINGS.containsKey(key)) {
40                  JSONObject settingsFile = readJSONObject();
41                  SETTINGS.put(key, settingsFile.getInt(key));
42                  return SETTINGS.get(key);
43              } else {
44  
45                  return SETTINGS.get(key);
46              }
47          } catch (JSONException | NullPointerException e) {
48              super.handleLoadingFailure(new JSONException(fileNameString()));
49              return -1;
50          }
51      }
52  
53      @Override
54      String filePathString() {
55          String fp = "mtd/json/";
56          return fp;
57      }
58  
59      @Override
60      String fileNameString() {
61          String fn = "settings.json";
62          return fn;
63      }
64  
65  }
