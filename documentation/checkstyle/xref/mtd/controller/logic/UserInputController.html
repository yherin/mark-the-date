18  /**
19   * UserInputController performs functions of the controller in the MVC model. It
20   * controls the GUI View and asks for data from Model (GameWindow and
21   * QuizMaster).
22   *
23   * @author Jack Sheridan
24   *
25   */
26  public class UserInputController {
27  
28      /**
29       * GUI view to which commands are given.
30       */
31      private final GameWindow gui;
32      /**
33       * Data model from which data is requested and given to.
34       */
35      private final QuizMaster model;
36  
37      /**
38       * Create a new UserInputController which automatically sets up
39       * AnswerButtonListeners and then listens for user input.
40       *
41       * @param gui view component
42       * @param model model component
43       */
44      
45      public UserInputController(GameWindow gui, QuizMaster model) {
46          this.gui = gui;
47          this.model = model;
48          addListenerButtonsToGUI();
49      }
50  
51      /**
52       * ActionListener implementation which listens for player pressing an answer
53       * button.
54       */
55      class AnswerButtonListener implements ActionListener {
56  
57          /**
58           * When AnswerButton is clicked: timer is stopped, score for the
59           * question is added to the total score, timer is reset. Then QuizMaster
60           * (Model) is asked to provide the next question.
61           *
62           * @param ActionEvent ae caused by clicking AnswerButton.
63           */
64          @Override
65          public void actionPerformed(ActionEvent ae) {
66              AnswerButton clicked = (AnswerButton) ae.getSource();
67              clicked.setEnabled(false);
68              gui.getTimer().stop();
69              model.addPointsToScore(clicked.getAnswer().getScore());
70              nextQuestion();
71              gui.getTimer().reset();
72              clicked.setEnabled(true);
73          }
74  
75          private int calculatePoints(Answer a) {
76              int baseScore = a.getScore();
77              int bonus = gui.getTimer().currentTime() * 50;
78              return baseScore + bonus;
79          }
80  
81      }
82  
83      /**
84       * ActionListener implementation which listeners for user pressing 'play
85       * again'.
86       */
87      class PlayAgainButtonListener implements ActionListener {
88  
89          @Override
90          public void actionPerformed(ActionEvent ae) {
91              createNewQuiz();
92              model.resetScore();
93              Question q = model.getSpecifiedQuestion(GUICommand.CURRENT_QUESTION);
94              gui.updateGUIToShowNextQuestion(q);
95          }
96  
97          private void createNewQuiz() {
98              model.buildNewQuiz();
99          }
100 
101     }
102 
103     private void addListenerButtonsToGUI() {
104         List<AnswerButtonListener> listeners = createListeners();
105         for (int i = 0; i < listeners.size(); i++) {
106             this.gui.addListenerToAnswerButton(listeners.get(i), i);
107         }
108         JButton playAgain = (JButton) GUIComponentMap.getComponentByEnum(GUIComponent.BUTTON_PLAY_AGAIN);
109         playAgain.addActionListener(new PlayAgainButtonListener());
110     }
111 
112     private List<AnswerButtonListener> createListeners() {
113         List<AnswerButtonListener> aBLL = new ArrayList<>();
114         for (int i = 0; i < 4; i++) {
115             aBLL.add(new AnswerButtonListener());
116         }
117         return aBLL;
118     }
119 
120     private void nextQuestion() {
121         Question current = this.model.getSpecifiedQuestion(GUICommand.CURRENT_QUESTION);
122         Question next = this.model.getSpecifiedQuestion(GUICommand.NEXT_QUESTION);
123         if (!current.equals(next)) {
124             this.gui.updateGUIToShowNextQuestion(next);
125         } else {
126             JTextArea score = (JTextArea) GUIComponentMap.getComponentByEnum(GUIComponent.LABEL_SCORE_INT);
127             score.setText(String.valueOf(this.model.getScore()));
128             this.gui.addSummaryComponents();
129         }
130 
131     }
132
