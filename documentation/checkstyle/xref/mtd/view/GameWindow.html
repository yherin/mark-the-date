11  import javax.swing.JFrame;
12  import javax.swing.JPanel;

14  import javax.swing.JTextField;
15  import mtd.model.command.QuizMaster;
16  import mtd.model.models.Question;
17  import mtd.model.models.QuestionStopwatch;
18  import mtd.model.models.answer.Answer;
19  import mtd.view.create.QuestionComponentCreator;
20  import mtd.view.create.ResultsComponentCreator;

22  /**
23   *
24   * @author Jack Sheridan
25   */
26  public class GameWindow implements Runnable {
27  
28      //core
29      private Container container;
30      private Question currentQuestion;
31      private JFrame frame;
32      private QuizMaster logic;
33  
34      //question view
35      private JTextArea questionText;
36      private List<AnswerButton> answerButtons;
37      private QuestionStopwatch timer;
38  
39      //summary view
40      private JTextArea summary;
41      private JButton quit;
42      private JButton newGame;
43  
44      //Utility
45      private ResultsComponentCreator summaryCreator;
46      private QuestionComponentCreator questionComponentCreator;
47  
48      public GameWindow(QuizMaster logic) {
49          assignMembers(logic);
50          createFrame();
51          createAndMapComponents();
52          assignCreatedComponentsToLocalVariables();
53          addQuestionComponents();
54          updateAnswerLabels(currentQuestion.getShuffled());
55          this.showGUI();
56  
57      }
58  
59      private void createFrame() {
60          frame = new JFrame("Mark the date");
61          frame.setSize(500, 500);
62          frame.setLayout(new GridBagLayout());
63          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
64          container = frame.getContentPane();
65      }
66  
67      private void assignMembers(QuizMaster logic) {
68          this.logic = logic;
69          this.currentQuestion = this.logic.getSpecifiedQuestion(GUICommand.CURRENT_QUESTION);
70      }
71  
72      private void assignCreatedComponentsToLocalVariables() {
73          //Question components
74          answerButtons = new ArrayList<>();
75          answerButtons.add((AnswerButton) GUIComponentMap.getComponentByEnum(GUIComponent.BUTTON_ANSWER_1));
76          answerButtons.add((AnswerButton) GUIComponentMap.getComponentByEnum(GUIComponent.BUTTON_ANSWER_2));
77          answerButtons.add((AnswerButton) GUIComponentMap.getComponentByEnum(GUIComponent.BUTTON_ANSWER_3));
78          answerButtons.add((AnswerButton) GUIComponentMap.getComponentByEnum(GUIComponent.BUTTON_ANSWER_4));
79          questionText = (JTextArea) GUIComponentMap.getComponentByEnum(GUIComponent.LABEL_QUESTION_TEXT);
80          timer = (QuestionStopwatch) GUIComponentMap.getComponentByEnum(GUIComponent.TIMER);
81  
82          //Summary components
83          this.quit = (JButton) GUIComponentMap.getComponentByEnum(GUIComponent.BUTTON_QUIT);
84          this.newGame = (JButton) GUIComponentMap.getComponentByEnum(GUIComponent.BUTTON_PLAY_AGAIN);
85          this.summary = (JTextArea) GUIComponentMap.getComponentByEnum(GUIComponent.LABEL_SUMMARY_TEXT);
86      }
87  
88      private void createAndMapComponents() {
89          questionComponentCreator = new QuestionComponentCreator();
90          questionComponentCreator.createAndMapComponents();
91          summaryCreator = new ResultsComponentCreator();
92          summaryCreator.createAndMapComponents();
93      }
94  
95      public void addListenerToAnswerButton(ActionListener al, int index) {
96          this.answerButtons.get(index).addActionListener(al);
97      }
98  
99      public final void updateGUIToShowNextQuestion(Question nextQuestion) {
100         updateAnswerLabels(nextQuestion.getShuffled());
101         updateQuestionText(nextQuestion.getEvent().getDescription());
102         timer.reset();
103         addQuestionComponents();
104         showGUI();
105     }
106 
107     public final void updateAnswerLabels(List<Answer> shuffledAnswers) {
108         for (int i = 0; i < shuffledAnswers.size(); i++) {
109             answerButtons.get(i).setAnswer(shuffledAnswers.get(i));
110         }
111 
112     }
113 
114     private void updateQuestionText(String text) {
115         JTextArea event = (JTextArea) GUIComponentMap.getComponentByEnum(GUIComponent.LABEL_QUESTION_TEXT);
116         event.setText(text);
117     }
118 
119     public final void addSummaryComponents() {
120         container.removeAll();
121         container.add(summary);
122         container.add(quit);
123         container.add(newGame);
124         showGUI();
125     }
126 
127     private void addQuestionComponents() {
128         container.removeAll();
129         container.add(questionText);
130         for (AnswerButton answerButton : answerButtons) {
131             container.add(answerButton);
132 
133         }
134         container.add(timer);
135     }
136 
137     private void clearComponents() {
138 
139     }
140 
141     @Override
142     public void run() {
143         this.showGUI();
144     }
145 
146     private void showGUI() {
147 
148         frame.setVisible(true);
149         frame.repaint();
150         this.timer.start();
151     }
152 
153     public QuestionStopwatch getTimer() {
154         return timer;
155     }
156 
157 }
